# 동전1

### [ 문제 정보 ]
> **백준**: https://www.acmicpc.net/problem/2293
> 
> **난이도**: 1
>
> **유형**: #동적계획법 (Dynamic Programming)


### [ 문제 풀이 ]
```C++
#include <bits/stdc++.h>
#include <cmath>
using namespace std;


int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n ,k;
    cin >> n >>  k;
    vector<long long> dp(k+1);
    dp[0] = 1;
    for (int i =1 ; i <=k ;i++) {
        dp[i] = 0;
    }

    for (int i=1; i<=n; i++) {
        int coin;
        cin >> coin;
        for ( int j = 1; j <= k ;j++) {
            if (j < coin) {
                dp[j] = dp[j];
            } else { // j >= coin
                dp[j] = dp[j-coin] + dp[j];
            }
        }
    }
    cout << dp[k] << endl;
}

```

```C++
// 이건 2차원으로 푼거
#include <bits/stdc++.h>
#include <cmath>
using namespace std;


int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n ,k;
    cin >> n >>  k;
    vector<vector<long long>> dp(n+1, vector<long long>(k+1) );
    for (int i =0 ; i <=k ;i++) {
        dp[0][i] = 0;
    }
    for ( int i= 0 ; i<=n; i++) {
        dp[i][0] =1;
    }
    for (int i=1; i<=n; i++) {
        int coin;
        cin >> coin;
        for ( int j = 1; j <= k ;j++) {
            if (j < coin) {
                dp[i][j] = dp[i-1][j];
            } else { // j >= coin
                dp[i][j] = dp[i][j-coin] + dp[i-1][j];
            }
        }
    }
    cout << dp[n][k] << endl;
}
```
