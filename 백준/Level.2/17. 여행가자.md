# 여행가자

### [ 문제 정보 ]
> **백준**: https://www.acmicpc.net/problem/1976
> 
> **난이도**: 2
>
> **유형**: #자료구조


### [ 문제 풀이 ]
```C++
#include <bits/stdc++.h>
#include <sstream>
using namespace std;


int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    long long n,m;
    cin >> n;
    cin >> m;

    map<int, set<int>> adjustedList;

    for (int i=1 ;i<=n; i++) {
        for (int j = 1; j<=n; j++) {
            bool isConnected ;
            cin >> isConnected;
            adjustedList[i].insert(i);
            if (isConnected) {
                adjustedList[i].insert(j);
            }
        }
        for ( set<int>::iterator iter = adjustedList[i].begin(); iter != adjustedList[i].end(); iter++) {
            adjustedList[*iter].insert(adjustedList[i].begin(), adjustedList[i].end());
        }
    }

    bool no = false;
    int startIndex =0;
    for (int i = 1; i<=m ; i++) {
        if (i==1) {
            cin >> startIndex;
        }else {
            int haveToCity;
            cin >> haveToCity;
            if (adjustedList[startIndex].find(haveToCity) == adjustedList[startIndex].end()) {
                cout << "NO"<<"\n";
                no = true;
                break;
            }
        }
    }
    if (!no)
        cout << "YES" ;



}
```
