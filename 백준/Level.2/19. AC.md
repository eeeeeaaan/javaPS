# AC

### [ 문제 정보 ]
> **백준**: https://www.acmicpc.net/problem/5430
> 
> **난이도**: 2
>
> **유형**: #구현, #자료구조


### [ 문제 풀이 ]
```C++
#include <iostream>
#include <algorithm>
#include <sstream>
#include <deque> 
#include <string>

using namespace std;

int main() {
    ios::sync_with_stdio(false); 
    cin.tie(NULL);

    int num;
    cin >> num; 

    while (num--) {
        string inst, array;
        int count;
        cin >> inst >> count >> array;

        deque<string> dq;
        if (count > 0) {
            string pureArray = array.substr(1, array.length() - 2);
            istringstream ss(pureArray);
            string token;
            while (getline(ss, token, ',')) {
                dq.push_back(token);
            }
        }

        bool isReversed = false;
        bool isError = false;

        // 2. 핵심 로직: erase 대신 방향성만 체크
        for (char cmd : inst) {
            if (cmd == 'R') {
                isReversed = !isReversed; 
            } else if (cmd == 'D') {
                if (dq.empty()) {
                    isError = true;
                    break;
                }
                if (isReversed) {
                    dq.pop_back(); 
                } else {
                    dq.pop_front(); 
                }
            }
        }

        // 3. 출력 로직
        if (isError) {
            cout << "error\n";
        } else {
            cout << "[";
            if (isReversed) {
                for (auto it = dq.rbegin(); it != dq.rend(); ++it) {
                    cout << *it << (next(it) == dq.rend() ? "" : ",");
                }
            } else {
                for (auto it = dq.begin(); it != dq.end(); ++it) {
                    cout << *it << (next(it) == dq.end() ? "" : ",");
                }
            }
            cout << "]\n";
        }
    }
    return 0;
}
```
처음엔 벡터를 썼으나 요소의 삭제시 시간 복잡도가 O(n)이 되어서 dequeue를 사용하는 것으로 변경
