# 가장 긴 바이토닉 부분 수열

### [ 문제 정보 ]
> **백준**: https://www.acmicpc.net/problem/11054
> 
> **난이도**: 2
>
> **유형**: #동적계획법 (Dynamic Programming)


### [ 문제 풀이 ]
```C++
#include <bits/stdc++.h>
#include <cmath>
using namespace std;


int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n ;
    cin >> n ;
    vector<long long> v(n+1);
    v[0]=0;
    for ( int i =1 ; i<=n ;i++) {
        cin >> v[i];
    }

    vector<int> increase_result(n+2);
    vector<int> decrease_result(n+2);

    increase_result[0] = 0;
    decrease_result[0] = 0;
    increase_result[n+1] = 0;
    decrease_result[n+1] = 0;
    for (int i = 1 ; i<=n ;i++) {
        int Max =0;
        for ( int j = 0 ; j<=i-1; j++) {
            if (v[j]< v[i] && increase_result[j] > Max)
                Max = increase_result[j];

        }
        increase_result[i] = Max+1;
    }

    for ( int i = n ;i>=1 ; i--) {
        int Max = 0 ;
        for (int j = n+1 ; j > i ;j--) {
            if (v[j]< v[i] && decrease_result[j] > Max)
                Max = decrease_result[j];
        }
        decrease_result[i] = Max+1;
    }
    int maxLength =0;
        for (int i = 1; i<=n; i++) {
            if (maxLength < increase_result[i]+ decrease_result[i]) {
                maxLength = increase_result[i]+ decrease_result[i];
            }
        }
        cout << maxLength -1 << endl;

}

```
