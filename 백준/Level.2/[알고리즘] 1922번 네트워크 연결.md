# 네트워크 연결

### [ 문제 정보 ]
> **백준**: https://www.acmicpc.net/problem/1922
> 
> **난이도**: 2
>
> **유형**: #최소신장트리 #Kruskal 


### [ 문제 풀이 ]
```C++
#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;
int parent[1000000];
int ranking[1000000];
class Edge {
public:
    int node[2];
    int distance;

    Edge(int n , int m, int distance) {
        node[0] = n ;
        node[1] = m ;
        this->distance = distance ;
    }

    bool operator <(Edge &e) {
        return this -> distance < e.distance;
    }
};

void initial(int n){
    for (int i=0 ; i< n;i++) {
        parent[i] = i ;
        ranking[i]= 1;
    }
}

int find(int x) {
    if (parent[x] != x) {
        parent[x] = find(parent[x]);
    }
    return parent[x];
}

void merge(int x, int y) {
    int rootX= find(x);
    int rootY= find(y);

    if (rootX == rootY)
        return ;

    if (ranking[rootX] < ranking[rootY]) {
        parent[rootX] = rootY;
    } else if (ranking[rootX] > ranking[rootY]) {
        parent[rootY] = rootX;
    } else {
        parent[rootX] = rootY;
        ranking[rootY] += 1;
    }
}

bool equal(int x, int y) {
    return find(x) == find(y);
}

const int INF = 100000001;
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);


    vector<Edge> v;
    int n, m ;
    cin >> n ;
    cin >> m ;
    for (int i=0 ; i<m;i++) {
        int a,b,c;
        cin >> a >> b >> c ;
        v.push_back(Edge(a,b,c));
    }

    sort(v.begin(), v.end());
    initial(n);
    long long count =0;
    vector<Edge> f; // mst 에 포함된 노드 저장
    for (int i=0; (i< v.size()) && (f.size() < n-1); i++ ) {
        int a = v[i].node[0];
        int b = v[i].node[1];
        int p = find(a);
        int q = find(b);

        if (!equal(p,q)) {
            merge(p,q);
            count += v[i].distance;
            f.push_back(v[i]);

        }
    }
    cout << count << endl ;
}
```
